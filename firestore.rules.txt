rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    /* ---------- Role helpers ---------- */
    // Reads the caller's own /users/{uid} doc. Firestore rules cache get()
    // calls within a single request, so this is cheap to call repeatedly.
    function myRole() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data;
    }
    function isSignedIn() {
      return request.auth != null;
    }
    // Signed in AND has an active role doc (owner/admin/staff).
    function isActiveUser() {
      return isSignedIn()
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && myRole().active == true;
    }
    function isOwner() {
      return isActiveUser() && myRole().role == "owner";
    }
    // Owner or Admin — full day-to-day management rights.
    function canManage() {
      return isActiveUser() && myRole().role in ["owner", "admin"];
    }

    // ---------- Team / role documents ----------
    // Every signed-in user may read their OWN role doc (needed to gate the
    // app after login). Only the owner can read the full team list, and
    // only the owner can create/edit/deactivate team members. The very
    // first-ever user is allowed to create their own "owner" doc once
    // (app-side bootstrap flow) as long as the team is still empty.
    match /users/{uid} {
      allow get: if isSignedIn() && (request.auth.uid == uid || isOwner());
      allow list: if isOwner();
      allow create: if isOwner()
                    || (isSignedIn() && request.auth.uid == uid
                        && request.resource.data.role == "owner"
                        && !exists(/databases/$(database)/documents/meta/teamBootstrap));
      allow update: if isOwner();
      allow delete: if false; // deactivate (active:false) instead of deleting
    }

    // One-time bootstrap marker — written once when the first owner account
    // is created, read by anyone signed in so they know whether bootstrap
    // has already happened.
    match /meta/teamBootstrap {
      allow read: if isSignedIn();
      allow create: if isSignedIn() && !exists(/databases/$(database)/documents/meta/teamBootstrap);
      allow update, delete: if false;
    }

    // Stock items — any active team member (owner/admin/staff) can read.
    // Only owner/admin can create, edit, or delete an item.
    match /stockItems/{itemId} {
      allow read: if isActiveUser();
      allow create, update, delete: if canManage();
    }

    // Stock in/out transaction log — read by any active team member.
    // Created by any active team member (staff log receive/issue too),
    // never edited or deleted (keeps an honest audit trail).
    match /stockTransactions/{txId} {
      allow read: if isActiveUser();
      allow create: if isActiveUser();
      allow update, delete: if false;
    }

    // Approval requests (discrepancies, new-item requests, threshold changes) —
    // readable/creatable by any active team member. Only owner/admin can
    // resolve (approve/reject) a request; a request can never be edited
    // back to "pending" or deleted once created.
    match /approvals/{approvalId} {
      allow read: if isActiveUser();
      allow create: if isActiveUser()
                    && request.resource.data.status == "pending"
                    && request.resource.data.type in ["discrepancy", "new-item", "threshold"];
      allow update: if canManage()
                    && request.resource.data.status in ["approved", "rejected"];
      allow delete: if false;
    }

    // Suppliers directory — read by any active team member (GRN needs it
    // for supplier suggestions), created/edited/deleted only by owner/admin.
    match /suppliers/{supplierId} {
      allow read: if isActiveUser();
      allow create, update, delete: if canManage();
    }

    // Goods Received Notes — the permanent record of every receipt.
    // Read/created by any active team member (staff receive stock too),
    // never edited or deleted once submitted (keeps an honest audit trail).
    match /grns/{grnId} {
      allow read: if isActiveUser();
      allow create: if isActiveUser();
      allow update, delete: if false;
    }

    // Sequence counters (GRN numbers, supplier codes, etc.) — read/write
    // by any active team member via a transaction.
    match /counters/{counterId} {
      allow read, write: if isActiveUser();
    }

    // Everything else stays locked down by default.
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
