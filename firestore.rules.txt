rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Stock items — the app is now locked behind sign-in, so only
    // signed-in ("added") users can read at all. Only a signed-in admin
    // (created by you in Firebase Auth) can add, edit, or delete an item.
    match /stockItems/{itemId} {
      allow read: if request.auth != null;
      allow create, update, delete: if request.auth != null;
    }

    // Stock in/out transaction log — read for the activity feed by any
    // signed-in user, create only by signed-in admins, never edited or
    // deleted (keeps an honest audit trail).
    match /stockTransactions/{txId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update, delete: if false;
    }

    // Approval requests (discrepancies, new-item requests, threshold changes) —
    // readable and creatable by any signed-in user. Only a signed-in admin
    // can resolve (approve/reject) a request, and a request can never be
    // edited back to "pending" or deleted once created.
    match /approvals/{approvalId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null
                    && request.resource.data.status == "pending"
                    && request.resource.data.type in ["discrepancy", "new-item", "threshold"];
      allow update: if request.auth != null
                    && request.resource.data.status in ["approved", "rejected"];
      allow delete: if false;
    }

    // Suppliers directory — read by any signed-in user (GRN needs it for
    // supplier suggestions), created/edited/deleted only by signed-in admins.
    match /suppliers/{supplierId} {
      allow read: if request.auth != null;
      allow create, update, delete: if request.auth != null;
    }

    // Goods Received Notes — the permanent record of every receipt.
    // Read by any signed-in user, created by signed-in admins, never
    // edited or deleted once submitted (keeps an honest audit trail).
    match /grns/{grnId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update, delete: if false;
    }

    // Sequence counters (GRN numbers, supplier codes, etc.) — read/write
    // by any signed-in user via a transaction; there is nothing sensitive
    // in a counter document itself.
    match /counters/{counterId} {
      allow read, write: if request.auth != null;
    }

    // Everything else stays locked down by default.
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
